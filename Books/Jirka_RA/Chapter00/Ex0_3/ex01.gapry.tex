\subsubsection*{Exercise 0.3.1. (Gapry)}
\begin{flalign*}
x \in (A \setminus (B \cap C)) &\implies (x \in A) \text{ and } (x \notin (B \cap C))                      &\\
                               &\implies (x \in A) \text{ and } (x \notin B \text{ or } x \notin C))       &\\
                               &\implies (x \in A \text{ and } x \notin B) \text{ or }
                                         (x \in A \text{ and } x \notin C) \text{ (distributive property)} &\\
                               &\implies (x \in A \setminus B) \text{ or } (x \in A \setminus C)           &\\
                               &\implies x \in ((A \setminus B) \cup (A \setminus C))                      &\\
% 
x \in ((A \setminus B) \cup (A \setminus C)) &\implies ((x \in A) \text{ and } (x \notin B)) \text{ or }
                                                       ((x \in A) \text{ and } (x \notin C))               &\\ 
                                             &\implies (x \in A) \text{ and } ((x \notin B) \text{ or } (x \notin C))
                                                       \text{ (distributive property)}                     &\\
                                             &\implies (x \in A) \text{ and } (x \notin (B \cap C))        &\\
                                             &\implies x \in (A \setminus (B \cap C))
\end{flalign*}
